for (let i = 0; i < numbers.length - 1; i++) {
  for (let j = 0; j < numbers.length - 1 - i; j++) {
    // もし左の数値と右の数値が左の数値が大きい場合、入れ替えを行う
    if (numbers[j] < numbers[j + 1]) {
      let saveBox = numbers[j];
      // 入れ替え。大きいデータを右に移動
      numbers[j] = numbers[j + 1];
      numbers[j + 1] = saveBox;
    }
  }
}
